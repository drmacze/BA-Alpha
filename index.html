"Login" : "Register"; document.querySelector("button").innerText = isLogin ? "Login" : "Register"; } function showMessage(msg, type = 'error') { errorMsg.innerText = ""; successMsg.innerText = ""; if (type === 'error') errorMsg.innerText = msg; else successMsg.innerText = msg; } function login() { const email = emailInput.value; const pass = passwordInput.value; if (!email || !pass) { showMessage("Please fill all fields."); return; } if (isLogin) { auth.signInWithEmailAndPassword(email, pass) .then(() => showMessage("Login successful!", "success")) .catch(err => showMessage(err.message)); } else { if (pass.length < 6) return showMessage("Password must be at least 6 characters."); auth.createUserWithEmailAndPassword(email, pass) .then(() => showMessage("Account created!", "success")) .catch(err => showMessage(err.message)); } } function forgotPassword() { const email = emailInput.value; if (!email) return showMessage("Enter your email to reset password."); auth.sendPasswordResetEmail(email) .then(() => showMessage("Reset link sent to your email.", "success")) .catch(err => showMessage(err.message)); } function googleLogin() { const provider = new firebase.auth.GoogleAuthProvider(); auth.signInWithPopup(provider) .then(() => showMessage("Google login successful!", "success")) .catch(err => showMessage(err.message)); } function guestLogin() { auth.signInAnonymously() .then(() => showMessage("Logged in as guest.", "success")) .catch(err => showMessage(err.message)); } auth.onAuthStateChanged(user => { if (user) { console.log("User logged in:", user.uid); // Redirect to main game if needed } });
